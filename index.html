Quick Installation

# Clone repository
git clone https://github.com/yourusername/HealthRisk-AI.git

# Change directory
cd HealthRisk-AI

# Install dependencies
pip install -r requirements.txt

# Setup database
python setup_database.py

# Run the application
streamlit run app.py
